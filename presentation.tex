\subsection{Spécificité du langage}
\begin{frame}[fragile]{\insertsection}
\begin{block}{\insertsubsection}
\begin{itemize}
\item \visible<1->{Langage interprété: script Matlab (.m)}
\item \visible<2->{Pas de déclaration de variables.}
\item \visible<3->{L'absence d'un point virgule indique que le résultat d'une instruction doit être affiché dans la fenêtre de commande.}
\item  \visible<4->{{\color{red}\danger}: Après l'execution d'un script, les variables restent en mémoire 
\begin{itemize}
\item L'instruction \texttt{clear all} permet de vider l'espace mémoire.
\end{itemize}}
\item  \visible<5->{{\color{red}\danger}: Le premier élément d'un tableau est indexé à 1 (et non 0).}
\item  \visible<6->{{\color{red}\danger}: Ne pas confondre:
\begin{itemize}
\item \texttt{a*b}: multiplication matricielle entre \texttt{a} et \texttt{b},
\item \texttt{a.*b}: multiplication terme à terme entre \texttt{a} et \texttt{b}.
\end{itemize}}
\end{itemize}
\end{block}
\end{frame}
